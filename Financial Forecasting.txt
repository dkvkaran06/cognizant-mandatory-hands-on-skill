Exercise 7: Financial Forecasting
public class FinancialForecast {

    public static double forecastRevenue(double currentRevenue,
                                         double growthRate,
                                         int years) {

        // Base case
        if (years == 0) {
            return currentRevenue;
        }

        // Recursive case
        return forecastRevenue(currentRevenue, growthRate, years - 1)
                * (1 + growthRate);
    }

    public static void main(String[] args) {

        double currentRevenue = 100000; // ₹100,000
        double growthRate = 0.10;       // 10%
        int years = 5;

        double futureRevenue =
                forecastRevenue(currentRevenue, growthRate, years);

        System.out.printf("Forecasted Revenue after %d years: ₹%.2f%n",
                          years, futureRevenue);
    }
}
